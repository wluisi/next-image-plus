# Picture Component Example

<PictureExampleTabs>
```tsx
import { Picture, Source, Img } from "next-image-plus";

export default function Card() {
  return (
    <div className="flex flex-col lg:flex-row overflow-hidden rounded-lg border border-gray-200">
      <div className="card-image relative lg:flex-shrink-0">
        <div className="relative aspect-[2/1] lg:aspect-square">
          <Picture preload={preload} fallbackMedia="(max-width: 430px)">
            <Source
              src="/images/large.jpg"
              width={600}
              height={600}
              media="(min-width: 960px)"
            />
            <Source
              src="/images/medium.jpg"
              width={400}
              height={200}
              media="(min-width: 600px) and (max-width:959px)"
            />
            <Img
              src="https://picsum.photos/id/59/430/215"
              width={600}
              height={600}
              alt="Mountains and a river"
            />
          </Picture>
        </div>
      </div>
    </div>
  );
}
```
